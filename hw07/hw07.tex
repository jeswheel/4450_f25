    \item \label{prob:cdf}(1 point) Suppose that $X$ is a discrete random variable, with cdf given by Table~\ref{tab:cdf}. Find $E[X]$ and $\Var(X)$.
    \begin{table}[ht]
    \centering
    \begin{tabular}{cr}\hline
    $x$ & $F(x)$ \\\hline
    0 & 0 \\
    1 & .1 \\
    2 & .3 \\
    3 & .7 \\
    4 & .8 \\
    5 & 1.0\\\hline
    \end{tabular}
    \caption{\label{tab:cdf}The cdf for $X$ in Problem~\ref{prob:cdf}}
    \end{table}

    \item (1 point) Suppose that $X$ follows a Poisson$(\lambda)$ distribution. Find $E\big[1 / (X + 1)\big]$.

    \item (1 point) Let $X$ have a Gamma$(\alpha, \lambda)$ distribution. For those values of $\alpha$ and $\lambda$ for which it is defined, find $E[1/X]$.

    % \item Find the density of the minimum of $n$ independent Weibull random variables, each of which has the density
    % $$
    % f(t) = \beta \alpha^{-\beta} t^{\beta - 1}e^{-(t/\alpha)^\beta}
    % $$

    \item (1 point) Suppose we have $n$ independent and identical samples from a population, denoted $X_1, X_2, \ldots, X_n$. Let $E[X_i] = \mu$ (which we assume exists, and is the same for all $i$).
      \begin{enumerate}
        \item Calculate the expected value of $\bar{X}_n$, the sample mean
    $$
    \bar{X}_n = \frac{1}{n} \sum_{i = 1}^n X_i
    $$
        \item Is $\bar{X}_n$ an \emph{unbiased} estimator of $\mu$? If not, find values of $a$ and $b$ such that the linear transformation such that $a\bar{X}_n + b$ is an unbiased estimate of $\mu$.
      \end{enumerate}

    \item (2 point) Using the same setup as above, suppose now that the variance of the population is also finite, i.e., $\Var(X_i) = \sigma^2 < \infty$.
      \begin{enumerate}
        \item Consider calculating the average squared deviance $Y$
        $$
        Y = \frac{1}{n}\sum_{i = 1}^n (X_i - \bar{X}_n)^2. 
        $$
        What is $E[Y]?$
        
        \item Is $Y$ an \emph{unbiased} estimator of $\sigma^2$? If not, find values of $a$ and $b$ such that the linear transformation $aY + b$ is an unbiased estimate of $\sigma^2$.
      \end{enumerate}
    
    \item (1 points) Let $X$ be an exponential$(\lambda)$ random variable, which has standard deviation $\sigma = 1/\lambda$. 
    Find 
    $$
    P\big(|X - E[X]| > k\sigma),
    $$
    for $k = 2, 3, 4$, and compare the results to the bounds from Chebyshev's inequality. 
    
    
    \item (2 points) Suppose a child randomly types the letters Q W E R T Y on a keyboard 1000 times.
    \begin{enumerate}
      \item What is the expected number of times that the sequence QQQQ appears (counting overlaps)?
      \item Can you provide an upper bound on the probability that the sequence QQQQ appears more than $n$ times? How many times $n$ for it to happen would you be surprised? 
    \end{enumerate}

\end{enumerate}
