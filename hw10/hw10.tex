{\large{\textbf{Homework 10 (Final Exam Review)}}}
\end{center}

\noindent \textbf{Note:} This homework will only be graded on submission.
It's purpose is to help prepare you for the final exam.

\begin{enumerate}
  \item (1.8.9, Rice) The weather forecaster says that the probability of rain on Saturday is 20\% and that the probability of rain on Sunday is 25\%. Is the probability of rain during the weekend 50\%? Why or why not?
  \item (1.8.12, Rice) In a game of poker, five players are each dealt 5 cards from a 52-card deck. How many ways are there to deal the cards?
  \item (2.5.39, Rice) The Cauchy cumulative distribution function is
  $$
  F(x) = \frac{1}{2} + \frac{1}{\pi} \tan^{-1}(x), x \in \R.
  $$
  \begin{enumerate}
    \item Show that this is a CDF.
    \item Find the corresponding density function.
    \item Find $x$ such that $P(X > x) = 0.1$. 
  \end{enumerate}
  \item (3.8.18, Rice) Let $X$ and $Y$ have the joint density function $f(x, y) = k(x - y)$, for $0 \leq y \leq x \leq 1$.
  \begin{enumerate}
    \item Sketch the region over which the density is positive and use it in determining limits of integration to answer the following questions.
    \item Find $k$.
    \item Find the marginal densities of $X$ and $Y$. 
    \item Find the conditional densities of $Y$ given $X$ and $X$ given $Y$.
  \end{enumerate}
  \item Let $X \sim  N(\mu, \sigma^2)$ and $Y = aX + b$. First using a change of variables approach (CDF method or change of variables formula), find the density of Y. Next, find the denisty of $Y$ by first considering the MGF of $Y$, and then matching that to the MGF of a known distribution.
  \item (3.8.69, Rice) Find the density of the minimum of $n$ independent Weibull random variables, each of which has the density
    $$
    f(t) = \beta \alpha^{-\beta} t^{\beta - 1}e^{-(t/\alpha)^\beta}.
    $$
  \item (4.7.25, Rice) If $X_1$ and $X_2$ are independent random variables following a gamma distribution with parameters $\alpha$ and $\lambda$, find $E(R^2)$, where $R = X_1^2 + X_2^2$.
  \item (4.7.49, Rice) Two independent measurements, $X$ and $Y$, are taken of a quantity $\mu$, such that $E[X] = E[Y] = \mu$. Denote $\Var(X) = \sigma^2_X$, and $\Var(Y) = \sigma^2_Y$. The two measurements are combined by means of a weighted average to give
  $$
  Z = \alpha X + (1 - \alpha) Y,
  $$
  where $0 \leq \alpha \leq 1$. 
  \begin{enumerate}
    \item Show that $E(Z) = \mu$.
    \item Find $\alpha$ in terms of $\sigma_X$ and $\sigma_Y$ to minimize $\Var(Z)$.
    \item Under what circumstances is it better to use the average $(X + Y)/2$ than either $X$ or $Y$ alone? 
  \end{enumerate}
  \item Let $Y$ have a density which is symmetric about $0$ and let $X = SY$, where $S$ is independent of $Y$ and assumes values $1$ and $-1$ with probability $1/2$. Show that Cov$(X, Y) = 0$ but that $X$ and $Y$ are {\it not} independent. (This shows that uncorrelatedness does not necessarily imply independence.)
  \item Let $(U, V)$ be distributed jointly with a spherically symmetric density. In other words, let their joint density $f(u, v) = Cg(u^2+v^2)$, for some non-negative function $g$. Show that $(\epsilon_1U, \epsilon_2V)$ has the same distribution as $(U, V)$, where $\epsilon_1$ and $\epsilon_2$ are either $1$ or $-1$. Deduce that $U$ and $V$ are uncorrelated.
  \item Let $(X,Y)$ have a joint uniform distribution inside the ellipse given by $\{(x,y): \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1\}$. Find the marginal densities of $X$ and $Y$ and the conditional densities of $Y$ given $X$ and $X$ given $Y$. Are $X$ and $Y$ correlated? Are they independent? 
  \item (5.4.15, Rice) Suppose that you bet \$5 on each of a sequence of 50 independent fair games.
  Use the central limit theorem to approximate the probability that you will lose more than \$75.
  \item Let $X_1, X_2, \ldots, X_n$ be iid random variables with $E[X_i] = 2$, and $\Var(X_i) = 4$. Using the CLT and the Delta-method, get an approximate distribution for $\big(\bar{X}_n\big)^{4/3}$ if $\bar{X}_n$ is the average of the variables $X_1, \ldots, X_n$.
  
\end{enumerate}
